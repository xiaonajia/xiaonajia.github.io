Q2：首字母大写

eg:已知有字符串foo=”get-element-by-id”,写一个function将其转化成驼峰表示法”getElementById”
					
	firstLetterToUpperCase("get-element-by-id");
	
	function firstLetterToUpperCase(str) {
		var str_arr = str.split("-"),a = [],b="",c="";
		for(var i = 0; i < str_arr.length; i++ ){
			b = str_arr[i].charAt(0).toUpperCase() + str_arr[i].substr(1,str_arr.length-1);
			c = (i==0) ? str_arr[i] : b;
			a.push(c);
		}
		return a.join("");
	}
